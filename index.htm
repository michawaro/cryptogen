-- Créer la table a_la_une pour stocker les vidéos "A la une !" CREATE TABLE IF NOT EXISTS a_la_une ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, user_id TEXT NOT NULL, video_index INTEGER NOT NULL CHECK (video_index >= 0 AND video_index <= 3), video_url TEXT, video_title TEXT, created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(), updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(), UNIQUE(user_id, video_index) ); -- Créer un index sur user_id pour améliorer les performances de recherche CREATE INDEX IF NOT EXISTS idx_a_la_une_user_id ON a_la_une(user_id); -- Créer un index sur video_index pour améliorer les performances CREATE INDEX IF NOT EXISTS idx_a_la_une_video_index ON a_la_une(video_index); -- Créer un index composite sur user_id et video_index (déjà couvert par UNIQUE mais utile pour les requêtes) CREATE INDEX IF NOT EXISTS idx_a_la_une_user_video ON a_la_une(user_id, video_index); -- Fonction pour mettre à jour automatiquement updated_at CREATE OR REPLACE FUNCTION update_updated_at_column() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at = NOW(); RETURN NEW; END; $$ LANGUAGE plpgsql; -- Trigger pour mettre à jour updated_at automatiquement -- Supprimer le trigger s'il existe déjà DROP TRIGGER IF EXISTS update_a_la_une_updated_at ON a_la_une; CREATE TRIGGER update_a_la_une_updated_at BEFORE UPDATE ON a_la_une FOR EACH ROW EXECUTE FUNCTION update_updated_at_column(); -- Activer Row Level Security (RLS) si nécessaire ALTER TABLE a_la_une ENABLE ROW LEVEL SECURITY; -- Politique RLS : permettre la lecture à tous CREATE POLICY "Allow public read access" ON a_la_une FOR SELECT USING (true); -- Politique RLS : permettre l'insertion et la mise à jour seulement si l'utilisateur correspond -- Note : Adaptez cette politique selon vos besoins de sécurité CREATE POLICY "Allow insert for authenticated users" ON a_la_une FOR INSERT WITH CHECK (true); CREATE POLICY "Allow update for authenticated users" ON a_la_une FOR UPDATE USING (true) WITH CHECK (true); CREATE POLICY "Allow delete for authenticated users" ON a_la_une FOR DELETE USING (true);
